### Rate Limit Handling
```javascript
async function rateLimitedRequest(fn, retries = 3) {
  for (let i = 0; i < retries; i++) {
    try {
      return await fn();
    } catch (error) {
      if (error.status === 429) {
        const waitTime = Math.pow(2, i) * 1000; // Exponential backoff
        await new Promise(r => setTimeout(r, waitTime));
      } else {
        throw error;
      }
    }
  }
  throw new Error("Rate limit exceeded after retries");
}
```

```python
        variation_ids = []
        for title in variations:
            new_post = await mcp_create_post(
                title=title,
                content=original["content"],
                status="draft",
                meta={"ab_test_original": post_id}
            )
            variation_ids.append(new_post["id"])
        
        # Track performance over time
        test = {
            "original_id": post_id,
            "variation_ids": variation_ids,
            "started_at": datetime.now().isoformat(),
            "metrics": {}
        }
        
        # Store test config
        await mcp_set_option(f"ab_test_{post_id}", json.dumps(test))
        
        return test
    
    async def check_test_results(self, test_id):
        """Check A/B test results"""
        
        test = json.loads(await mcp_get_option(f"ab_test_{test_id}"))
        
        # Collect metrics for each variation
        results = {}
        for post_id in [test["original_id"]] + test["variation_ids"]:
            results[post_id] = {
                "views": await self.get_post_views(post_id),
                "ctr": await self.get_post_ctr(post_id),
                "time_on_page": await self.get_time_on_page(post_id)
            }
        
        # Determine winner
        winner = max(results, key=lambda x: results[x]["ctr"])
        
        return {
            "test_id": test_id,
            "results": results,
            "winner": winner,
            "confidence": self.calculate_confidence(results)
```
